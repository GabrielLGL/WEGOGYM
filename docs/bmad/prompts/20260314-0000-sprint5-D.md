## Implémentation

### Type de métrique
```ts
type ChartMetric = 'weight' | 'reps' | 'orm' | 'volume'
```

### État local à ajouter dans ExerciseHistoryContent
```ts
const [chartMetric, setChartMetric] = useState<ChartMetric>('weight')
```

### Données du graphique dynamiques (remplacer `chartData` existant)
```ts
const chartData = useMemo(() => {
  if (chartStats.length < 2) return null

  const getData = (stat: ExerciseSessionStats): number => {
    switch (chartMetric) {
      case 'weight': return stat.maxWeight
      case 'reps':   return stat.maxReps
      case 'orm':    return Math.round(stat.maxWeight * (1 + stat.maxReps / EPLEY_FORMULA_DIVISOR))
      case 'volume': return stat.totalVolume
    }
  }

  return {
    labels: chartStats.map(s =>
      new Date(s.date).toLocaleDateString(dateLocale, { day: 'numeric', month: 'short' })
    ),
    datasets: [{ data: chartStats.map(getData) }],
  }
}, [chartStats, chartMetric, dateLocale])
```
