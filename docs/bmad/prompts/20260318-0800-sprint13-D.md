## Implementation

### exerciseFrequencyHelpers.ts

```ts
export interface ExerciseFrequencyEntry {
  exerciseId: string
  exerciseName: string
  muscles: string[]
  count: number           // nombre de seances ou l'exercice apparait
  lastPerformed: Date | null
  daysSinceLastPerformed: number | null
  trend: 'increasing' | 'decreasing' | 'stable'
}

export interface ExerciseFrequencyResult {
  entries: ExerciseFrequencyEntry[]
  totalExercisesUsed: number
  mostFrequent: string | null
  leastFrequent: string | null
  neglected: ExerciseFrequencyEntry[]  // exercices pas faits depuis > 30j
}

/**
 * Calcule la frequence de chaque exercice.
 *
 * @param sets — tous les sets
 * @param exercises — tous les exercices
 * @param histories — toutes les seances
 * @param periodDays — 30, 90, null
 *
 * Algorithme :
 * 1. Pour chaque exercice, compter le nombre de histories uniques ou il apparait
 * 2. Trouver la date du dernier set pour chaque exercice
 * 3. Trend : comparer count 1ere moitie vs 2e moitie de la periode
 * 4. Trier par count decroissant
 * 5. neglected = exercices avec lastPerformed > 30j ou null
 */
export function computeExerciseFrequency(
  sets: Array<{ exerciseId: string; historyId: string; createdAt: Date | number }>,
  exercises: Array<{ id: string; name: string; muscles: string[] }>,
  histories: Array<{ id: string; startedAt: Date | number; isAbandoned: boolean }>,
  periodDays: number | null,
): ExerciseFrequencyResult
```
