## Implementation

### volumeDistributionHelpers.ts

```ts
export interface VolumeDistributionEntry {
  muscle: string
  totalVolume: number       // kg total (weight * reps)
  totalSets: number
  percentage: number        // 0-100 du total
}

export interface VolumeDistributionResult {
  entries: VolumeDistributionEntry[]
  totalVolume: number
  period: string
  dominantMuscle: string | null
  weakestMuscle: string | null
  balanceScore: number  // 0-100, 100 = parfaitement equilibre
}

/**
 * Calcule la distribution du volume par groupe musculaire.
 *
 * @param sets — sets filtres par periode
 * @param exercises — tous les exercices (pour mapper exerciseId -> muscles)
 * @param periodDays — nombre de jours (7, 30, 90, null pour tout)
 *
 * Algorithme :
 * 1. Pour chaque set, trouver l'exercice correspondant
 * 2. Distribuer le volume (weight * reps) sur chaque muscle de l'exercice
 *    (divise equitablement si multi-muscle)
 * 3. Sommer par muscle
 * 4. Calculer les pourcentages
 * 5. Trier par volume decroissant
 * 6. balanceScore = 100 - (ecart-type des pourcentages * 5), clamp [0, 100]
 */
export function computeVolumeDistribution(
  sets: Array<{ weight: number; reps: number; exerciseId: string; createdAt: Date | number }>,
  exercises: Array<{ id: string; muscles: string[] }>,
  periodDays: number | null,
): VolumeDistributionResult
```
