## Implementation

### muscleBalanceHelpers.ts

```ts
export interface MusclePair {
  name: string           // ex: 'Push / Pull'
  left: { label: string; volume: number }   // ex: Pecs + Epaules + Triceps
  right: { label: string; volume: number }  // ex: Dos + Biceps
  ratio: number          // left/right, ideal = 1.0
  status: 'balanced' | 'slight_imbalance' | 'imbalanced'
}

export interface MuscleBalanceResult {
  pairs: MusclePair[]
  overallBalance: number  // 0-100
  periodDays: number
}

/**
 * Calcule l'equilibre entre groupes musculaires agonistes/antagonistes.
 *
 * Paires analysees :
 * 1. Push / Pull → (Pecs + Epaules + Triceps) vs (Dos + Biceps)
 * 2. Quads / Ischios → Quadriceps vs Ischios
 * 3. Avant-bras → Biceps vs Triceps (ratio ideal ~0.7)
 * 4. Haut / Bas → (Pecs+Dos+Epaules+Biceps+Triceps+Trapezes) vs (Quads+Ischios+Mollets)
 *
 * Status :
 * - ratio 0.8-1.2 → 'balanced'
 * - ratio 0.6-0.8 ou 1.2-1.5 → 'slight_imbalance'
 * - sinon → 'imbalanced'
 *
 * Filtrable par periode (30, 90, tout).
 */
export function computeMuscleBalance(
  sets: Array<{ weight: number; reps: number; exerciseId: string; createdAt: Date | number }>,
  exercises: Array<{ id: string; muscles: string[] }>,
  periodDays: number | null,
): MuscleBalanceResult
```

**Barre comparative :**
```tsx
<View style={styles.pairCard}>
  <Text style={styles.pairName}>{pair.name}</Text>
  <View style={styles.pairBars}>
    <View style={styles.barSide}>
      <Text style={styles.barLabel}>{pair.left.label}</Text>
      <View style={[styles.barFill, { width: leftWidth, backgroundColor: colors.primary }]} />
    </View>
    <Text style={[styles.ratioText, { color: getStatusColor(pair.status, colors) }]}>
      {pair.ratio.toFixed(2)}
    </Text>
    <View style={styles.barSide}>
      <View style={[styles.barFill, { width: rightWidth, backgroundColor: '#8B5CF6' }]} />
      <Text style={styles.barLabel}>{pair.right.label}</Text>
    </View>
  </View>
  <View style={[styles.statusBadge, { backgroundColor: getStatusColor(pair.status, colors) }]}>
    <Text style={styles.statusText}>{t.stats.muscleBalance.statuses[pair.status]}</Text>
  </View>
</View>
```

**Couleurs status :**
- balanced → '#10B981'
- slight_imbalance → '#F59E0B'
- imbalanced → colors.danger
