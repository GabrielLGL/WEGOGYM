## Implementation

### setQualityHelpers.ts

```ts
export interface SetQualityEntry {
  exerciseId: string
  exerciseName: string
  totalSets: number
  avgWeight: number
  weightVariance: number        // ecart-type du poids (plus bas = plus consistant)
  avgReps: number
  repConsistency: number        // 0-100 (100 = toutes les series memes reps)
  dropSetsDetected: number      // nombre de drop sets detectes
  qualityScore: number          // 0-100 composite
  grade: 'A' | 'B' | 'C' | 'D' // note globale
}

export interface SetQualityResult {
  entries: SetQualityEntry[]
  overallScore: number          // moyenne des qualityScore
  overallGrade: 'A' | 'B' | 'C' | 'D'
  mostConsistent: string | null
  leastConsistent: string | null
}

/**
 * Analyse la qualite et consistance des series par exercice.
 *
 * @param sets — tous les sets
 * @param exercises — tous les exercices
 * @param periodDays — 30, 90, null (tout)
 *
 * Algorithme :
 * 1. Grouper sets par exerciseId
 * 2. Pour chaque exercice (min 5 sets) :
 *    a. avgWeight = moyenne des poids
 *    b. weightVariance = ecart-type / avgWeight * 100 (coefficient de variation)
 *    c. repConsistency = 100 - (ecart-type reps / avgReps * 100), cap 0-100
 *    d. dropSetsDetected = sequences ou poids diminue entre sets consecutifs d'une meme history
 *    e. qualityScore = repConsistency * 0.5 + (100 - weightVariance) * 0.3 + (totalSets > 20 ? 20 : totalSets) * 0.2
 * 3. Grade : >= 80 → A, >= 60 → B, >= 40 → C, < 40 → D
 * 4. Trier par qualityScore desc
 */
export function computeSetQuality(
  sets: Array<{ weight: number; reps: number; exerciseId: string; historyId: string; createdAt: Date | number }>,
  exercises: Array<{ id: string; name: string }>,
  periodDays: number | null,
): SetQualityResult
```
