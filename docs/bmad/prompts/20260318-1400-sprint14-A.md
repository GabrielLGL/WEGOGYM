## Implementation

### workoutSummaryHelpers.ts

```ts
export interface WorkoutSummary {
  sessionName: string
  programName: string | null
  date: Date
  durationMinutes: number
  totalVolume: number        // kg
  totalSets: number
  totalReps: number
  density: number            // kg/min
  prsHit: number             // nombre de PRs dans cette seance
  topExercise: string | null // exercice avec le plus gros volume
  timeAgo: string            // "il y a 2h", "hier", "il y a 3j"
}

/**
 * Construit le resume de la derniere seance.
 *
 * @param histories — toutes les seances non-abandonnees
 * @param sets — tous les sets
 * @param exercises — tous les exercices
 * @param sessions — toutes les sessions
 * @param programs — tous les programmes
 *
 * Algorithme :
 * 1. Trouver la derniere history (tri par startedAt desc, !isAbandoned)
 * 2. Filtrer les sets de cette history
 * 3. totalVolume = somme(weight * reps)
 * 4. durationMinutes = (completedAt - startedAt) / 60000
 * 5. density = totalVolume / durationMinutes (min 1)
 * 6. prsHit : compter les sets ou weight est > max precedent pour cet exercice
 * 7. topExercise : exercice avec le plus gros volume dans la seance
 * 8. timeAgo : formater la difference entre maintenant et startedAt
 * 9. Remonter session.name et program.name via les relations
 */
export function computeWorkoutSummary(
  histories: Array<{ id: string; startedAt: Date | number; completedAt: Date | number | null; isAbandoned: boolean; sessionId: string }>,
  sets: Array<{ weight: number; reps: number; exerciseId: string; historyId: string; createdAt: Date | number }>,
  exercises: Array<{ id: string; name: string }>,
  sessions: Array<{ id: string; name: string; programId: string }>,
  programs: Array<{ id: string; name: string }>,
): WorkoutSummary | null

/**
 * Formate "il y a Xh", "hier", "il y a Xj"
 */
export function formatTimeAgo(date: Date, language: 'fr' | 'en'): string
```

### Section UI dans HomeScreen

```tsx
{lastWorkout && (
  <View style={styles.lastWorkoutCard}>
    <View style={styles.lastWorkoutHeader}>
      <Text style={styles.lastWorkoutTitle}>{t.home.lastWorkout.title}</Text>
      <Text style={styles.lastWorkoutTime}>{lastWorkout.timeAgo}</Text>
    </View>
    <Text style={styles.lastWorkoutSession}>

      {lastWorkout.programName && ` — ${lastWorkout.programName}`}

    <View style={styles.lastWorkoutStats}>
      <View style={styles.statItem}>
        <Text style={styles.statValue}>{lastWorkout.durationMinutes}m</Text>
        <Text style={styles.statLabel}>{t.home.lastWorkout.duration}</Text>
      </View>
      <View style={styles.statItem}>
        <Text style={styles.statValue}>{(lastWorkout.totalVolume / 1000).toFixed(1)}t</Text>
        <Text style={styles.statLabel}>{t.home.lastWorkout.volume}</Text>
      </View>
      <View style={styles.statItem}>
        <Text style={styles.statValue}>{lastWorkout.totalSets}</Text>
        <Text style={styles.statLabel}>{t.home.lastWorkout.sets}</Text>

  </View>
```
