## Implementation

### trainingDensityHelpers.ts

```ts
export interface DensityResult {
  currentDensity: number      // kg/min actuellement
  elapsedMinutes: number
  totalVolume: number         // kg total de la seance en cours
  previousDensity: number | null  // densite de la derniere fois pour cette session
  comparison: 'faster' | 'slower' | 'similar' | null
  pace: string                // ex: "125 kg/min"
}

/**
 * Calcule la densite d'entrainement en temps reel.
 *
 * @param currentSets — sets de la seance en cours
 * @param startTime — timestamp de debut de seance
 * @param previousSets — sets de la derniere seance du meme programme (optionnel)
 * @param previousDuration — duree de la derniere seance en minutes (optionnel)
 *
 * Algorithme :
 * 1. totalVolume = somme(weight * reps) pour currentSets
 * 2. elapsedMinutes = (now - startTime) / 60000, min 1
 * 3. currentDensity = totalVolume / elapsedMinutes
 * 4. Si previousSets et previousDuration :
 *    previousDensity = somme(weight*reps des prev) / previousDuration
 *    comparison : diff > 10% → 'faster', diff < -10% → 'slower', sinon 'similar'
 */
export function computeTrainingDensity(
  currentSets: Array<{ weight: number; reps: number }>,
  startTime: Date | number,
  previousSets?: Array<{ weight: number; reps: number }>,
  previousDuration?: number,
): DensityResult

/**
 * Formate la densite : "125 kg/min"
 */
export function formatDensity(density: number): string
```

### Section UI dans WorkoutScreen

Petit badge/chip sous le header, a cote du timer si present.

```tsx
{densityData && densityData.totalVolume > 0 && (
  <View style={styles.densityBadge}>
    <Text style={styles.densityValue}>{formatDensity(densityData.currentDensity)}</Text>
    {densityData.comparison && (
      <Text style={[styles.densityComparison, {
        color: densityData.comparison === 'faster' ? '#10B981' : densityData.comparison === 'slower' ? colors.danger : colors.textSecondary
      }]}>
        {densityData.comparison === 'faster' ? '↑' : densityData.comparison === 'slower' ? '↓' : '→'}
      </Text>
    )}
  </View>
)}
```

**Styles :**
