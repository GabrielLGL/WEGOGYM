## Implementation

### volumeRecordsHelpers.ts

```ts
export interface VolumeRecord {
  type: 'session' | 'week' | 'month'
  recordVolume: number       // kg
  recordDate: Date
  currentVolume: number      // volume de la periode en cours
  percentOfRecord: number    // 0-100+
  isNewRecord: boolean       // currentVolume > recordVolume
}

export interface VolumeRecordsResult {
  records: VolumeRecord[]
  totalLifetimeVolume: number
  avgSessionVolume: number
  avgWeeklyVolume: number
  recentTrend: 'up' | 'down' | 'stable'  // tendance sur 4 semaines
}

/**
 * Detecte les records de volume a differentes echelles.
 *
 * @param histories — seances non-abandonnees
 * @param sets — tous les sets
 *
 * Algorithme :
 * 1. Session record :
 *    - Pour chaque history, calculer le volume total (somme weight*reps des sets)
 *    - Trouver le max
 *    - currentVolume = volume de la derniere seance
 * 2. Week record :
 *    - Grouper par semaine (lundi-dimanche)
 *    - Volume par semaine = somme des volumes de toutes les seances de la semaine
 *    - Trouver le max
 *    - currentVolume = volume de la semaine en cours
 * 3. Month record :
 *    - Grouper par mois
 *    - Volume par mois = somme des volumes du mois
 *    - Trouver le max
 *    - currentVolume = volume du mois en cours
 * 4. recentTrend : comparer volume moyen des 2 dernieres semaines vs 2 precedentes
 *    - diff > 10% → 'up', diff < -10% → 'down', sinon → 'stable'
 */
export function computeVolumeRecords(
  histories: Array<{ id: string; startedAt: Date | number; isAbandoned: boolean }>,
  sets: Array<{ weight: number; reps: number; historyId: string; createdAt: Date | number }>,
): VolumeRecordsResult
```

### StatsVolumeRecordsScreen.tsx

**Structure :**
- Header avec volume total lifetime
- 3 cartes record (session, semaine, mois)
- Tendance recente

**Carte record :**
```tsx
<View style={styles.recordCard}>
  <View style={styles.recordHeader}>
    <Text style={styles.recordType}>{getRecordLabel(record.type, t)}</Text>
    {record.isNewRecord && (
      <View style={styles.newRecordBadge}>
        <Text style={styles.newRecordText}>{t.stats.volumeRecords.newRecord}</Text>
      </View>
    )}
  </View>
  <Text style={styles.recordVolume}>
    {(record.recordVolume / 1000).toFixed(1)}t
  </Text>
  <Text style={styles.recordDate}>
    {formatRecordDate(record.recordDate)}
  </Text>
  <View style={styles.recordBarTrack}>
    <View style={[styles.recordBarFill, {
      width: `${Math.min(record.percentOfRecord, 100)}%`,
      backgroundColor: record.percentOfRecord >= 100 ? '#10B981' : colors.primary,
    }]} />
  </View>
  <Text style={styles.recordCurrent}>
    {t.stats.volumeRecords.current}: {(record.currentVolume / 1000).toFixed(1)}t
    ({record.percentOfRecord}%)
  </Text>
</View>
```
