- Chaque ecran a une version `*Base` non-wrapped par withObservables — c'est celle a tester
- Pattern : voir `StatsConstellationScreen.test.tsx` comme reference

## Pattern de test

```tsx
import React from 'react'
import { render } from '@testing-library/react-native'
import { SomeScreenBase } from '../SomeScreen'

jest.mock('expo-haptics', () => ({
  impactAsync: jest.fn(),
  notificationAsync: jest.fn(),
  ImpactFeedbackStyle: { Light: 'Light', Medium: 'Medium', Heavy: 'Heavy' },
  NotificationFeedbackType: { Success: 'Success', Error: 'Error' },
}))

jest.mock('../../model/index', () => ({
  database: { write: jest.fn(), get: jest.fn() },
}))

// Si SVG utilise :
jest.mock('react-native-svg', () => ({
  Svg: 'Svg', Circle: 'Circle', Line: 'Line', Rect: 'Rect', Text: 'Text', G: 'G', Path: 'Path', Polygon: 'Polygon',
}))
```
