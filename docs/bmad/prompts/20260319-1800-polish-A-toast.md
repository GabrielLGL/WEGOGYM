### Spec du composant Toast

```tsx
// Types
type ToastVariant = 'success' | 'error' | 'info'

interface ToastConfig {
  message: string
  variant?: ToastVariant  // default: 'success'
  duration?: number       // default: 2500ms
}

// Hook API
const { showToast } = useToast()
showToast({ message: 'Programme sauvegarde', variant: 'success' })
showToast({ message: 'Erreur de suppression', variant: 'error' })
```

### Spec visuelle
- Position : bas de l'ecran (bottom: 100px pour eviter le footer)
- Animation : fade in (200ms) + slide up (translateY), fade out (300ms)
- Icone : Ionicons (`checkmark-circle` success, `close-circle` error, `information-circle` info)
- Couleur fond : `colors.card` avec bordure gauche coloree (success/danger/amber)
- Auto-dismiss apres `duration` ms
- Un seul toast visible a la fois (le nouveau remplace l'ancien)

### Implementation

**ToastContext.tsx :**
```tsx
import React, { createContext, useContext, useState, useCallback, useRef, useEffect } from 'react'
import { Toast } from '../components/Toast'

interface ToastContextValue {
  showToast: (config: ToastConfig) => void
}

const ToastContext = createContext<ToastContextValue>({ showToast: () => {} })

export function ToastProvider({ children }: { children: React.ReactNode }) {
  const [config, setConfig] = useState<ToastConfig | null>(null)
  const timerRef = useRef<ReturnType<typeof setTimeout>>()

  const showToast = useCallback((c: ToastConfig) => {
    if (timerRef.current) clearTimeout(timerRef.current)
    setConfig(c)
    timerRef.current = setTimeout(() => setConfig(null), c.duration ?? 2500)
  }, [])

  useEffect(() => () => { if (timerRef.current) clearTimeout(timerRef.current) }, [])

  return (
    <ToastContext.Provider value={{ showToast }}>
      {children}
      {config && <Toast config={config} onDismiss={() => setConfig(null)} />}
    </ToastContext.Provider>
  )
}

export const useToast = () => useContext(ToastContext)
```
