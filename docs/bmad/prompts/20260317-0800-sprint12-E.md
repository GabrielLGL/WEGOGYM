# Rapport — Progression Mensuelle (#110) — Groupe E — 20260317-0800

## Objectif

## Implementation

### monthlyProgressHelpers.ts

```ts
export interface MonthlyStats {
  month: string            // 'YYYY-MM'
  monthLabel: string       // 'Mars 2026'
  totalVolume: number      // kg total
  sessionCount: number
  prCount: number          // nombre de PRs ce mois
  activeDays: number       // jours uniques avec seance
  avgSessionsPerWeek: number
  topExercise: { name: string; volume: number } | null
  totalSets: number
  avgVolumePerSession: number
}

export interface MonthlyProgressResult {
  current: MonthlyStats
  previous: MonthlyStats
  deltas: {
    volume: number          // pourcentage de changement
    sessions: number
    prs: number
    activeDays: number
    avgPerWeek: number
    totalSets: number
  }
  trend: 'up' | 'down' | 'stable'  // basé sur volume delta
}

/**
 * Calcule les stats pour un mois donne et le mois precedent.
 *
 * @param histories — toutes les seances (non abandonnees)
 * @param sets — tous les sets
 * @param performanceLogs — logs de performance (pour detecter PRs)
 * @param exercises — map exerciceId -> nom
 * @param targetMonth — 'YYYY-MM' du mois cible (defaut : mois courant)
 *
 * Algorithme :
 * 1. Filtrer histories et sets pour le mois cible et le mois precedent
 * 2. Pour chaque mois :
 *    - totalVolume = somme(weight * reps) pour tous les sets
 *    - sessionCount = nombre de histories
 *    - activeDays = nombre de dates uniques
 *    - avgSessionsPerWeek = sessionCount / nombre de semaines dans le mois
 *    - topExercise = exercice avec le plus de volume total
 *    - prCount = nombre de performanceLogs avec is_pr === true dans la periode
 * 3. Calculer les deltas en pourcentage :
 *    - Si previous = 0 et current > 0 → +100%
 *    - Si previous = 0 et current = 0 → 0%
 *    - Sinon → ((current - previous) / previous) * 100
 * 4. trend : delta volume > 5% → 'up', < -5% → 'down', sinon 'stable'
 */
export function computeMonthlyProgress(
  histories: Array<{ id: string; startedAt: Date | number; isAbandoned: boolean }>,
  sets: Array<{ weight: number; reps: number; historyId: string; createdAt: Date | number }>,
  performanceLogs: Array<{ isPr: boolean; createdAt: Date | number }>,
  exercises: Array<{ id: string; name: string }>,
  targetMonth?: string,
): MonthlyProgressResult

/**
 * Retourne les mois disponibles (qui ont au moins 1 seance).
 */
export function getAvailableMonths(
  histories: Array<{ startedAt: Date | number; isAbandoned: boolean }>,
): string[]

/**
 * Formate le label du mois en francais/anglais.
 * 'YYYY-MM' → 'Mars 2026' (fr) / 'March 2026' (en)
 */
export function formatMonthLabel(yearMonth: string, language: 'fr' | 'en'): string
```

```tsx
function DeltaCard({ label, current, previous, delta, unit }: DeltaCardProps) {
  const isPositive = delta > 0
  const isNeutral = delta === 0
  return (
    <View style={styles.deltaCard}>
      <Text style={styles.deltaLabel}>{label}</Text>
      <Text style={styles.deltaValue}>{current}{unit}</Text>
      <View style={styles.deltaRow}>
        <Text style={[
          styles.deltaChange,
          { color: isNeutral ? colors.textMuted : isPositive ? '#10B981' : colors.danger }
        ]}>
          {isPositive ? '+' : ''}{delta.toFixed(0)}%
        </Text>
        <Text style={styles.deltaPrevious}>vs {previous}{unit}</Text>
      </View>
    </View>
  )
}
```
